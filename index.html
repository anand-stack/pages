Parsing Fasta Files

Fasta file is common file format in bioinformatics to represet sequences.
Here, it is my parser implementation to parse fasta file. 


            // fasta.hpp
            #include <vector>
            #include <string>

            struct Fasta;
            std::vector<string> read_fasta_file(std::string);
          

            // fasta.cpp
            #include <vector>
            #include <fstream>
            #include <string>

            struct Fasta {
              std::string header,sequence;
            };

            std::vector<string> read_fasta_file(std::string filename)
            {
              std::vector<string> fasta_records;
              std::ifstream file(filename);
              while(file)
              {
                std::string head,seq;
                Fasta record;
                getline(file,head);
                getline(file,seq);
                record.header = head;
                record.sequence = seq;
                fasta_records.push_back(record);
              }
              return fasta_records;
            }

Here Main program 


            // main.cpp
            #include <iostream>
            #include <vector>
            #include <string>

            using namespace std;
            using Fasta_Records = vector<Fasta>;

            int main(int argc, char** argv)
            {
              if(argc < 2) cout << "Usage :" << argv[0] << endl;
              else
              {
                Fasta_Records records = read_fasta_file(argv[1]);
                for (auto [head,seq] : records) cout << head << "\n" << seq << endl;
              }
            }
